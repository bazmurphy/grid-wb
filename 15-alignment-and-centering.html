.container {
  height: 500px;
  display: grid;
  grid-template-columns: repeat(4, 1fr);
  grid-template-rows: (2, 200px);

  /* justify is row axis */
  /* align is column axis */

  /* justify-items: stretch (default) | end  | center | stretch | auto  */
  justify-items: center;

  /* align-items: stretch (default) | end | center | stretch | space-between | space-around | space-evenly | baseline */
  align-items: center;

  /* shorthand for both^ */
  /* place-items: center center; */

  /* justify-content: start (default) | end | center | stretch | space-between | space-around | space-evenly */
  justify-content: space-between;

  /* align-content: stretch (default) | end | center | stretch | space-between | space-around | space-evenly */
  align-content: space-evenly;

  /* shorthand for both^ */
  /* place-content: space-evenly space-evenly; */

  gap: 20px;
}
