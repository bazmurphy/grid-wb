.grid1 {
  display: grid;
  grid-template-columns: repeat(12, 1fr);
  gap: 20px;
}

.grid2 {
  display: grid;
  grid-template-columns: repeat(12, 1fr);
  gap: 20px;
}

.grid2 .item {
  /* we can use min-width: 0 on the grid item */
  min-width: 0;
}

.grid3 {
  display: grid;
  /* we can use minmax(0, 1fr) */
  grid-template-columns: repeat(12, minmax(0, 1fr));
  gap: 20px;
}

.grid3 .item {
  /* and then use use width: 100% on the grid item */
  width: 100%;
}

.grid4 {
  display: grid;
  /* we can recieve the number of columns from the inline styling, with a fallback of 12 */
  grid-template-columns: repeat(var(--cols, 12), minmax(0, 1fr));
  gap: 20px;
}

.grid4 .item {
  min-width: 0;
  width: 100%;
  /* use css variable and overwrite the variable with inline styling */
  --span: 1;
  grid-column: span var(--span);
}
